Assembly program to add two numbers:

.model small
.stack 100h
.data
	a dw 2
	b dw 3
	res dw ?
.code
   main proc
	mov ax, a
	add ax, b
	mov res, ax
	mov ah, 4ch
	int 21h
   main endp
end main

      again:

Assembly program to print message hello:
.model small
.stack 100h
.data
	message db "Hello$"
.code 
  main proc
	mov ax, @data
	mov ds, ax

	mov ah, 09h
	lea dx, message
	int 21h

	mov ah, 4ch
	int 21h
  main endp
end main
